\section{{\Large The course of parametric complexity}}

\begin{frame}[fragile]{\thesection. Parametric PDEs and FEONets}
\framesubtitle{$\quad$}
{\Large {\color{blue}Equation:}} {\Large Given $ a (x), b (x), c (x) $, input parameter $ f (x) $, output solution $ u $:}
\begin{equation}
    - \text{div} (a (x) \nabla u) + b (x) \cdot \nabla u + c (x) u = f (x) \quad \text{in } D, \quad u|_{\partial D} = g (x).
\end{equation}
{\Large {\color{blue}Objective:} {\color{red}Estimate the error between the true solution and the FEONets solution.}}
\medskip

%{\Large {\color{blue}Question:} How does operator learning method work?}
%\medskip

{\Large {\color{blue} 1. Supervised learning (DeepONet, FNO):} $ \| u_{\text{NN}} (f; \theta) - u_{\text{trad}} \|^2 $}
\medskip

{\Large $ u_{\text{trad}} $ is the solution obtained by traditional numerical methods, e.g., FEM {\b$ A u_{\text{trad}} = L $}
\begin{align}
    \| u_{\text{NN}} (f;\theta) - A^{-1} L \|^2
\end{align} }

{\Large {\color{blue} 2. Unsupervised learning (FEONet):} $ \| A u_{\text{NN}}(f;\theta) - L \|^2 $}

\end{frame}

\begin{frame}{\thesection. FEONets}
\framesubtitle{$\quad$}
{\Large {\b$ f (x, \omega) = \omega_1 \sin{2 \pi \omega_2 x} + \omega_3 \cos{2 \pi \omega_4 x} $} }
\medskip

{\Large {\b Input}: $ \omega = (\omega_1, \omega_2, \omega_3, \omega_4) $, i.i.d. uniform distribution on $ \Omega:= [0,1]^4 $, \big({\b More general input}: $ \{f (x_i)\}_{i=1}^{N} $\big)} 
\medskip

{\Large {\b Solution structure}: $ {u}_{h,\theta} (x, \omega) = \sum_{k=1}^{N_h} {\alpha}_k (\omega) \phi_k (x) $, $ \{\phi_k\}_{k=1}^{N_h} $ is the finite element basis}
\medskip

{\Large {\b Output}: coefficients $ {\alpha}(\omega) := ({\alpha}_1 (\omega), {\alpha}_2 (\omega), \dots, {\alpha}_{N_h} (\omega)) $ 
}
\medskip

{\Large {\b Loss function}: $ \mathscr{L} ({\alpha}) := \int_{\Omega} \| A {\alpha} (\omega) - L \|_{\ell^2} \d P (\omega) $, $ A $ is the stiffness matrix, $ L $ is the load vector.}
\medskip

{\Large {\b Monte Carlo approximation}: $ \mathscr{L}^M ({\alpha}) = \frac{|\Omega|}{M} \sum_{j=1}^{M} \| A {\alpha} (\omega^j) - L \|_{\ell^2} $, $ \{\omega^j\}_{j=1}^{M} $ are i.i.d. samples from $ \Omega $}
\end{frame} 

\begin{frame}{\thesection. Neural network architecture}
    \framesubtitle{$\quad$}
    {\Large {\b Neural network architecture}: For each $ L \in \mathbb{N} $, an $ L $-layer ReLU neural netwok: $ f^L (x) : \mathbb{R}^{n_0} \to \mathbb{R}^{n_L} $ is defined by
    \begin{align}
        f^1 (x) = W^1 x + b^1, \quad f^\ell (x) = W^\ell \sigma (f^{\ell-1} (x)) + b^\ell, \quad \ell = 2,3,\dots,L,
    \end{align} 
    where $ W^\ell \in \mathbb{R}^{n_\ell \times n_{\ell-1}} $, $ b^\ell \in \mathbb{R}^{n_\ell} $, and $ \sigma (x) = \max\{0,x\} $ is the ReLU activation function.
    }
    \medskip

    {\Large {\b Universal approximation theorem}: Suppose there is a sequence of NN spaces $ \{\mathcal{N}_{n}\}_{n=1}^{\infty} $ such that $ \mathcal{N}_n \subset \mathcal{N}_{n+1} $. Let $ K $ be a compact set in $ \mathbb{R}^{m} $ and $ g \in C (K, \mathbb{R}^{N}) $. Then there holds
    \begin{align}
        \lim_{n \to \infty} \inf_{\hat{g} \in \mathcal{N}_n} \| \hat{g} - g \|_{C (K)} = 0.
    \end{align} 
    }
\end{frame} 

\section{{\Large The Hamilton-Jacobi equation}}

\begin{frame}{\thesection. Analytic framework for convergence}
    \framesubtitle{$\quad$}
    {\Large {\b Exact FEM solution}: $ u_h (x,\omega) = \sum_{k=1}^{N_h} \alpha_k^* (\omega) \phi_k (x) $, i.e., $ A \alpha^* (\omega) = L $.}
    \medskip

{\Large {\b approximation without quadrature}: $ {\alpha}_{n}^{\mathscr{L}} := \arg\min_{\alpha \in \mathcal{N}_n} \mathscr{L} (\alpha) = \arg\min_{\alpha \in \mathcal{N}_n} \int_{\Omega} \| A \alpha (\omega) - L \|_{\ell^2} \d P (\omega) $
\begin{align}
    u_{h,n} (x,\omega) = \sum_{k=1}^{N_h} ({\alpha}_{n}^{\mathscr{L}})_k (\omega) \phi_k (x).
\end{align} 
}

{\Large {\b Final approximation}: $ {\alpha}_{n,M}^{\mathscr{L}} := \arg\min_{\alpha \in \mathcal{N}_n} \mathscr{L}^M (\alpha) = \arg\min_{\alpha \in \mathcal{N}_n} \frac{|\Omega|}{M} \sum_{j=1}^{M} \| A \alpha (\omega^j) - L \|_{\ell^2} $
\begin{align}
    u_{h,n,M} (x,\omega) = \sum_{k=1}^{N_h} ({\alpha}_{n,M}^{\mathscr{L}})_k (\omega) \phi_k (x).
\end{align} 
}
    
\end{frame} 

\begin{frame}{\thesection. Analytic framework for convergence}
    \framesubtitle{$\quad$}
    {\Large $ \alpha^* \sim u_{h}, \quad  {\alpha}_{n}^{\mathscr{L}} \sim u_{h,n} , \quad {\alpha}_{n,M}^{\mathscr{L}} \sim u_{h,n,M} $.}
    \medskip

    {\Large {\b Error decomposition}: $ u - u_{h,n,M} = (u - u_h) + (u_h - u_{h,n}) + (u_{h,n} - u_{h,n,M}) $.
\begin{align}
   \| u_h - u_{h,n,M} \|_{L^1 (\Omega; L^2 (D))} =  &\; \int_{\Omega} \| \sum_{j=1}^{N_h} (\alpha_j^* - ({\alpha}_{n,M}^{\mathscr{L}})_j ) \phi_j \|_{L^2 (D)} \d P (\omega) \notag \\
   \le &\; \int_{\Omega} \sum_{j=1}^{N_h} |\alpha_j^* - (\hat{\alpha}_{n,M}^{\mathscr{L}})_j| \| \phi_j \|_{L^2 (D)} \d P (\omega) \notag \\
   \le &\; C N_h^{\frac{1}{2}} \| \alpha^* - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \max_{1 \le j \le N_h} \| \phi_j \|_{L^2 (D)} \notag \\
   \le &\; C \| \alpha^* - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \notag \\
   \le &\; C \big( \| \alpha^* - {\alpha}_{n}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} + \| {\alpha}_{n}^{\mathscr{L}} - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \big).
\end{align} 
}
    
\end{frame} 

\begin{frame}{\thesection. Estimate of $ \| \alpha^* - {\alpha}_n^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} $}
    \framesubtitle{$\quad$}
    {\Large {\b $ \rho_{\text{min}}, \rho_{\text{max}} $}: the minimum and maximum singular values of $ A $.
    \begin{align}
        \| \alpha^* - {\alpha}_n^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \le &\; \frac{1}{\rho_{\text{min}}} \| A \alpha^* - A {\alpha}_n^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} = \frac{1}{\rho_{\text{min}}} \| A {\alpha}_n^{\mathscr{L}} - L \|_{L^1 (\Omega; \ell^2)} = \frac{1}{\rho_{\text{min}}} \mathscr{L} ({\alpha}_n^{\mathscr{L}}) \notag \\
        = &\; \frac{1}{\rho_{\text{min}}} \inf_{\alpha \in \mathcal{N}_n} \| A \alpha - L \|_{L^1 (\Omega; \ell^2)} \le \frac{\rho_{\text{max}}}{\rho_{\text{min}}} \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^1 (\Omega; \ell^2)} \notag \\
        = &\; \kappa (A) \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^1 (\Omega; \ell^2)} \to 0 \quad \text{as } n \to \infty,
    \end{align} 
    where $ \kappa (A) = \frac{\rho_{\text{max}}}{\rho_{\text{min}}} $ is the condition number of $ A $.
    }
    
\end{frame} 

\begin{frame}{\thesection. Estimate of $ \| {\alpha}_n^{\mathscr{L}} - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} $}
    \framesubtitle{$\quad$}
    {\Large {\b Rademacher complexity}: For a function class $ \mathscr{F} $ and a sample $ S = \{X_1, X_2, \dots, X_M\} $, the empirical Rademacher complexity is defined as
    \begin{align}
        \mathscr{R}_M (\mathscr{F}) = \mathbb{E}_{\{X_i\}_{i=1}^{M}} \mathbb{E}_{\{\varepsilon\}_{i=1}^{M}} \bigg[ \sup_{f \in \mathscr{F}} \frac{1}{M}\sum_{i=1}^{M} \varepsilon_i f (X_i) \bigg],
    \end{align} 
    where $ \varepsilon_i $ are i.i.d. Rademacher random variables, $ \mathbb{P} (\varepsilon_i = 1) = \mathbb{P} (\varepsilon_i = -1) = \frac{1}{2} $.
    \begin{align}
        \mathbb{E}_{\{X_i\}_{i=1}^{M}} \bigg[ \sup_{f \in \mathscr{F}} \Big| \frac{1}{M} \sum_{i=1}^{M} f(X_i) - \mathbb{E}_{X \sim \mathbb{P}_{\Omega}}[f(X)] \Big| \bigg] \le 2 \mathscr{R}_M (\mathscr{F}).
    \end{align} 
{\b Estimate}: $ \mathbb{E} \big[|\mathscr{L}^M(\alpha) - \mathscr{L}(\alpha)|\big] \le 2 \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) $, where $ \mathscr{F}_n^{\mathscr{L}} = \{ \|A \alpha - L\|_{\ell^2} : \alpha \in \mathcal{N}_n^* \} $.
    }

    
\end{frame}     

\begin{frame}{\thesection. Estimate of $ \| {\alpha}_n^{\mathscr{L}} - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} $ (continued)}
    \framesubtitle{$\quad$}
    {\Large 
    {\setlength{\abovedisplayskip}{-30pt}
    \begin{align}
        \mathbb{E} \big[ \| {\alpha}_n^{\mathscr{L}} - {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \big] \le &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\| A {\alpha}_n^{\mathscr{L}} - A {\alpha}_{n,M}^{\mathscr{L}} \|_{L^1 (\Omega; \ell^2)} \big] \notag \\
        \le &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\| A {\alpha}_n^{\mathscr{L}} - L \|_{L^1 (\Omega; \ell^2)} + \| A {\alpha}_{n,M}^{\mathscr{L}} - L \|_{L^1 (\Omega; \ell^2)}\big] \notag \\
        = &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\mathscr{L} ({\alpha}_{n}^{\mathscr{L}}) + \mathscr{L}({\alpha}_{n,M}^{\mathscr{L}})\big] \notag \\
        \le &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\mathscr{L}({\alpha}_n^{\mathscr{L}}) + \mathscr{L}^M({\alpha}_{n,M}^{\mathscr{L}})\big] + \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\mathscr{L}({\alpha}_{n,M}^{\mathscr{L}}) - \mathscr{L}^M({\alpha}_{n,M}^{\mathscr{L}})\big] \notag \\
        \le &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\mathscr{L}({\alpha}_n^{\mathscr{L}}) + \mathscr{L}^M({\alpha}_{n}^{\mathscr{L}})\big] + \frac{2}{\rho_{\text{min}}} \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) \notag \\
        \le &\; \frac{1}{\rho_{\text{min}}} \mathbb{E} \big[\mathscr{L}^M ({\alpha}_n^{\mathscr{L}}) - \mathscr{L} ({\alpha}_n^{\mathscr{L}})\big] + \frac{2}{\rho_{\text{min}}}\mathbb{E} \big[\mathscr{L} ({\alpha}_n^{\mathscr{L}})\big] + \frac{2}{\rho_{\text{min}}} \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) \notag \\
        \le &\; \frac{4}{\rho_{\text{min}}} \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) + 2  \kappa (A) \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{1}(\Omega; \ell^2)} .
    \end{align} }
    }

\end{frame} 

\begin{frame}{\thesection. Final result}
    \framesubtitle{$\quad$}
    {\Large {\b Final result}: If for all $ n \in \mathbb{N} $, $ \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) \to 0 $ as $ M \to \infty $, then
    \begin{align}
        \mathbb{E} [\| u_h - u_{h,n,M} \|_{L^1 (\Omega; L^2 (D))} ] \le C \big(\frac{1}{\rho_{\text{min}}} \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) + \kappa (A) \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{1}(\Omega; \ell^2)} \big) \to 0 \notag \\
         \quad \text{as } n, M \to \infty.
    \end{align} 
    }
	\medskip 
	
	{\Large {\b Explicit convergence rate}: For nonsymmetric case, no direct estimate for $ \frac{1}{\rho_{\text{min}}} $.}
    
\end{frame} 



\section{{\Large Explicit error for self-adjoint PDEs}}

\begin{frame}{\thesection. Explicit error for self-adjoint PDEs}
    \framesubtitle{$\quad$}
	{\Large {\b Self-adjoint case}: $ A $ is symmetric and positive definite, i.e., $ \rho_{\text{min}} = \lambda_{\text{min}} (A) \ge C h^d $, $ \rho_{\text{max}} = \lambda_{\text{max}} \le C h^{d-2} $.
\begin{align}
	\kappa (A) = \frac{\rho_{\text{max}}}{\rho_{\text{min}}} = \frac{\lambda_{\text{max}} (A)}{\lambda_{\text{min}} (A)} \le C h^{-2}.
\end{align} 
	}
	\medskip

	{\Large Still need to estimate {\b$ \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) $} and {\b$ \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{1}(\Omega;\ell^2)} $} to obtain the explicit convergence rate. 
	}
\end{frame}

\begin{frame}{ \thesection. Estimate of $ \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{1}(\Omega;\ell^2)} $}
    \framesubtitle{$\quad$}
    {\Large 
	\begin{Theorem}[Barron (1993); E et al. (2022)]
		Let $ g \in \mathscr{W} (\Omega) $ and $ n \in \mathbb{N} $. Then there exists a two-layer ReLU neural network $ g_{\theta} \in \mathcal{N}_n $ with the parameter {\b$ \theta = (a_j , b_j , c_j)_{j=1}^{n} $} and $ \mathcal{N}_n $ collecting {\b$ \frac{1}{n}\sum_{j=1}^n a_j \sigma (b_j \cdot \omega + c_j) _{i=1}^{n} $} such that {\b$ \| \theta \|_{\mathscr{P}} \le 2 \| g \|_{\mathscr{B}^2 (\Omega)} $}, and 
		\begin{align}
						\| g_{\theta} - g \|_{L^2 (\Omega;\ell^2)} \le C \| g \|_{\mathscr{B}^2 (\Omega;\ell^2)} n^{-\frac{1}{2}}.
		\end{align} 
	\end{Theorem} 
    $ \| \theta \|_{\mathscr{P}} := \big\|\frac{1}{n} \sum_{j=1}^n a_j \big(\| b_j \|_{\ell^1} + |c_j| \big)\big\|_{\ell^{\infty}}. $

    $ \| g \|_{\mathscr{B}^s(\Omega)} := \inf_{g_e |_{\Omega} = g} \int_{\mathbb{R}^{m}}(1+ |\xi|)^s |\hat{g}_e (\xi)| \d \xi  $, $ \hat{g}_e $ is the Fourier transform of $ g_e $     
    {\b\begin{align}
        \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{1}(\Omega;\ell^2)} \le C \inf_{\alpha \in \mathcal{N}_n} \| \alpha - \alpha^* \|_{L^{2} (\Omega;\ell^2)} \le C n^{-\frac{1}{2}} \| \alpha^* \|_{\mathscr{B}^2 (\Omega;\ell^2)}.
    \end{align} }
    }
\end{frame} 

\begin{frame}{\thesection. Estimate of $ \| \alpha^* \|_{\mathscr{B}^2 (\Omega)} $}
    \framesubtitle{$\quad$}
    {\Large Since $ \alpha^* (\omega) = A^{-1} L (\omega) $ and $ L = \int_{D} f (x,\omega) \Phi (x) \d x $ with $ \Phi = (\phi_1 , \phi_2 , \cdots , \phi_{N_h})^\top $, we have 
    \begin{align}
        \hat{\alpha}^* (\xi) = A^{-1} \int_{D} \hat{f} (x,\xi) \Phi (x) \d x.
    \end{align} 
    Then we have
    \begin{align}
        \| \alpha^* \|_{\mathscr{B}^2 (\Omega;\ell^2)} = &\; \int_{\mathbb{R}^{m}} (1 + |\xi|)^2 \| \hat{\alpha}^* (\xi) \|_{\ell^2} \d \xi \notag \\
        \le &\; \int_{\mathbb{R}^{m}} (1 + |\xi|)^2 \int_{D} |\hat{f} (x,\xi)| \|A^{-1} \Phi (x) \|_{\ell^2} \d x \d \xi .
    \end{align} 
    Since $ \| A^{-1} \Phi \|_{\ell^2} \le \frac{1}{\rho_{\text{min}}} \| \Phi (x) \|_{\ell^2} \le \frac{N_h^{1/2}}{ \rho_{\text{min}}} \sim h^{-3d/2} $, we have
    \begin{equation}
        \| \alpha^* \|_{\mathscr{B}^2 (\Omega;\ell^2)} \le C h^{-3d/2} \int_{D}\int_{\mathbb{R}^{m}} (1 + |\xi|)^2  |\hat{f} (x,\xi)| \d x \d \xi \sim {\b h^{-3d/2} \| f \|_{L^{1}(D;\mathscr{B}^2 (\Omega))}}.
    \end{equation} 
    }
    
\end{frame} 

\begin{frame}{\thesection. Estimate of $ \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) $}
    \framesubtitle{$\quad$}
    {\Large 
    \begin{Theorem}[Neyshabur et al., 2015; E et al., 2022]
        For the function class $ \mathcal{N}_{n}^{*} = \{g_{\theta} \in \mathcal{N}_n : \| \theta \|_{\mathscr{P}} \le Q \} $, we have
        \begin{align}
            \mathscr{R}_M (\mathcal{N}_n^*) \le C \frac{Q N_h}{\sqrt{M}}.
        \end{align}
    \end{Theorem} 
    %$ \mathcal{N}_n^* = \{\alpha \in \mathcal{N}_n : \| \theta \|_{\mathscr{P}} \le Q \}, \quad \mathscr{F}_{n}^{\mathscr{L}} = \{ \| A \alpha - L \|_{\ell^2}: \alpha \in \mathcal{N}_n^* \}$
    {\setlength{\abovedisplayskip}{-25pt}
    \begin{align}
        \mathscr{R}_M (\mathscr{F}_n^{\mathscr{L}}) = &\; \mathbb{E}_{\{X_i\}_{i=1}^{M}} \mathbb{E}_{\{\varepsilon\}_{i=1}^{M}} \bigg[ \sup_{\alpha \in \mathcal{N}_n^*} \frac{1}{M}\sum_{i=1}^{M} \varepsilon_i  \| A \alpha (\omega^i) - L \|_{\ell^2} \bigg] \notag \\
        \le &\; \mathbb{E}_{\{\omega^i\}_{i=1}^{M}} \mathbb{E}_{\{\varepsilon\}_{i=1}^{M}} \bigg[ \sup_{\alpha \in \mathcal{N}_n^*} \frac{1}{M}\sum_{i=1}^{M} \varepsilon_i  \| A \alpha (\omega^i) \|_{\ell^2} \bigg] \notag \\
        \le &\; \rho_{\text{max}} \mathbb{E}_{\{\omega^i\}_{i=1}^{M}} \mathbb{E}_{\{\varepsilon\}_{i=1}^{M}} \bigg[ \sup_{\alpha \in \mathcal{N}_n^*} \frac{1}{M}\sum_{i=1}^{M} \varepsilon_i  \| \alpha (\omega^i) \|_{\ell^2} \bigg] = \rho_{\text{max}} \mathscr{R}_M (\mathcal{N}_n^*) \le C \frac{Q}{\sqrt{M}} h^{-2}.
    \end{align} }
    } 
    
\end{frame} 

\begin{frame}{\thesection. Final convergence rate}
    \framesubtitle{$\quad$}
    {\Large {\b Final convergence rate}: If $ \| f \|_{L^{1}(D;\mathscr{B}^2 (\Omega))} < \infty $, then
    \begin{align}
        \mathbb{E} \Big[\| u - u_{h,n,M} \|_{L^1 (\Omega; L^2 (D))} \Big] \le &\; \mathbb{E} \Big[ \| u - u_h \|_{L^1 (\Omega; L^2 (D))} \Big] + C n^{-\frac{1}{2}} h^{-\frac{3d}{2}} + C Q h^{-2} M^{-\frac{1}{2}}  \notag \\
        \le &\; C h^{\ell + 1} + C n^{-\frac{1}{2}} h^{-\frac{3d}{2}} + C Q h^{-2} M^{-\frac{1}{2}} \notag \\
        = &\; C h^{\ell+1} + C n^{-\frac{1}{2}} \kappa (A)^{\frac{3d}{4}} + C Q \kappa (A) M^{-\frac{1}{2}}.
    \end{align} 
    }
    
\end{frame}

\section{{\Large Conclusion} }
\begin{frame}{\thesection. Conclusion}
\framesubtitle{$\quad$}
{\Large {\b Equation:} $ 
    - \text{div} (a (x) \nabla u) + b (x) \cdot \nabla u + c (x) u = f (x) \quad \text{in } D, \quad u|_{\partial D} = g (x).
 $} 
\medskip

{\Large {\b Solution Structure:} $ u = \sum_{i = 1}^{N_h}(\hat{\alpha}_{n,M}^{\mathscr{L}})_i \phi_i (t,x) $}
\medskip

{\Large {\b Loss function:} $ \mathscr{L}^M (\alpha) =  \frac{|\Omega|}{M} \sum_{j=1}^{M} \| A \alpha (\omega^j) - L \|_{\ell^2}  $}
\medskip

{\Large {\b Convergence rate:} $ C h^{\ell+1} + C n^{-\frac{1}{2}} \kappa (A)^{\frac{3d}{4}} + C Q \kappa (A) M^{-\frac{1}{2}} $}
\medskip

{\Large {\b Limitations}:
\begin{enumerate}
    \item Obtain the explicit convergence rate for nonsymmetric case
    \item Obtain the convergence rate for more general input, e.g., $ \{f (x_i)\}_{i=1}^{N} $
\end{enumerate} 
\medskip

{\b Future work}: nonlinear PDEs, time-dependent PDEs, complex geometries, etc.
}
\end{frame}
